<script>

const CONFIG = {

  brand: "Leveo",

  whatsappE164: "5581992288619",

  apiEndpoint: "https://piewydyoisyfqndokzbn.functions.supabase.co/leads",

  OPEN_WHATSAPP_ON_SUBMIT: true,

};

const digits = (s) => String(s||"").replace(/\D/g,"");

const buildWa = (msg) => `https://wa.me/${CONFIG.whatsappE164}?text=${encodeURIComponent(msg)}`;

const getUTM = () => {

  const p = new URLSearchParams(location.search);

  const keys = ["utm_source","utm_medium","utm_campaign","utm_term","utm_content"];

  const o = {}; keys.forEach(k => o[k] = p.get(k) || ""); return o;

};

const idemKey = (nome, wpp, email) => {

  const seed = `${nome}|${digits(wpp)}|${email||""}|${new Date().toISOString().slice(0,10)}`;

  let h=0; for(let i=0;i<seed.length;i++) h=(h*31+seed.charCodeAt(i))>>>0; return `lead-${h}`;

};

document.getElementById("duvidaTop").href = buildWa("Oi! Tenho uma dúvida rápida.");

document.getElementById("duvidaMid").href = buildWa("Oi! Tenho uma dúvida rápida.");

document.getElementById("suporte").href = buildWa("Oi! Preciso de suporte.");

document.getElementById("year").textContent = new Date().getFullYear();


document.getElementById("leadForm").addEventListener("submit", async (e) => {

  e.preventDefault();

  const nome = document.getElementById("nome").value.trim();

  const whatsapp = document.getElementById("whatsapp").value;

  const email = document.getElementById("email").value.trim();

  if (nome.length < 2) { alert("Por favor, informe seu nome."); return; }

  if (digits(whatsapp).length < 10) { alert("Precisamos de um WhatsApp válido com DDD."); return; }


  const msg = `Oi! Sou ${nome}. Quero o teste de 14 dias da ${CONFIG.brand}. Meu WhatsApp: ${digits(whatsapp)}.`; // e-mail não vai para o WhatsApp

  const waUrl = buildWa(msg);

  const waTab = CONFIG.OPEN_WHATSAPP_ON_SUBMIT ? window.open("about:blank", "_blank") : null;


  try {

    const payload = {

      nome, whatsapp: digits(whatsapp), email,

      createdAt: new Date().toISOString(),

      source: "landing-leveo",

      utm: getUTM(),

      idempotencyKey: idemKey(nome, whatsapp, email),

    };

    const resp = await fetch(CONFIG.apiEndpoint, {

      method: "POST",

      headers: { "Content-Type": "application/json", "X-Idempotency-Key": payload.idempotencyKey },

      body: JSON.stringify(payload),

    });

    if (!resp.ok) throw new Error("Falha ao salvar lead");

    document.getElementById("formWrap").classList.add("hidden");

    document.getElementById("success").classList.remove("hidden");

  } catch (err) {

    console.error(err);

    alert("Não consegui registrar seus dados agora, mas vou abrir o WhatsApp para falarmos.");

  } finally {

    if (CONFIG.OPEN_WHATSAPP_ON_SUBMIT) {

      if (waTab) waTab.location.href = waUrl; else window.open(waUrl, "_blank");

    }

  }

});
